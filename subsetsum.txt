import java.util.*;

public class SubsetSum {
    static boolean subsetSum(int[] arr, int n, int target, int index, int currentSum, ArrayList<Integer> subset) {
        if (currentSum == target) {
            System.out.println("Subset found: " + subset);
            return true;
        }
        if (index == n || currentSum > target)
            return false;
        subset.add(arr[index]);
        if (subsetSum(arr, n, target, index + 1, currentSum + arr[index], subset))
            return true;
        subset.remove(subset.size() - 1);
        if (subsetSum(arr, n, target, index + 1, currentSum, subset))
            return true;
        return false;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Enter number of elements:");
        int n = sc.nextInt();
        int[] arr = new int[n];
        System.out.println("Enter elements:");
        for (int i = 0; i < n; i++)
            arr[i] = sc.nextInt();
        System.out.println("Enter target sum:");
        int target = sc.nextInt();
        ArrayList<Integer> subset = new ArrayList<>();
        boolean found = subsetSum(arr, n, target, 0, 0, subset);
        if (found)
            System.out.println("Subset with given sum exists");
        else
            System.out.println("Subset with given sum does not exist");
        sc.close();
    }
}
